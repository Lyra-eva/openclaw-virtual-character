## 🚨 问题 1: 初始化时机不明确

### 当前问题

```typescript
// AvatarView.tsx - 组件挂载时初始化
const initializeUltimateService = async () => {
  await ultimateVoiceService.initialize();
};

useEffect(() => {
  initializeUltimateService();
}, []);

// 问题：组件卸载再挂载会重复初始化！
```

### 修复方案

**方案 A: 在应用入口初始化 (推荐)**

```typescript
// src/main.tsx
import { ultimateVoiceService } from './services/UltimateVoiceService';

async function initApp() {
  try {
    // 1. 先初始化语音服务
    await ultimateVoiceService.initialize({
      enableMicroExpressions: true,
      enableEmotionalTTS: true,
      enablePredictiveLipSync: true,
      enableBodyLanguage: true,
      enableContextAware: true,
    });
    console.log('✅ Voice services initialized');
  } catch (error) {
    console.error('❌ Voice init failed:', error);
    // 继续渲染应用，降级到基础模式
  }
  
  // 2. 渲染应用
  const app = document.getElementById('app');
  createRoot(app!).render(<App />);
}

initApp();
```

**方案 B: 使用懒加载 + 单例**

```typescript
// src/services/VoiceServiceSingleton.ts
import { ultimateVoiceService } from './UltimateVoiceService';

class VoiceServiceManager {
  private static instance: VoiceServiceManager;
  private initialized = false;
  private initPromise: Promise<void> | null = null;
  
  private constructor() {}
  
  static getInstance(): VoiceServiceManager {
    if (!VoiceServiceManager.instance) {
      VoiceServiceManager.instance = new VoiceServiceManager();
    }
    return VoiceServiceManager.instance;
  }
  
  async initialize(): Promise<void> {
    if (this.initialized) return;
    if (this.initPromise) return this.initPromise;
    
    this.initPromise = ultimateVoiceService.initialize();
    await this.initPromise;
    this.initialized = true;
  }
  
  isInitialized(): boolean {
    return this.initialized;
  }
}

export const voiceServiceManager = VoiceServiceManager.getInstance();
```

**使用：**
```typescript
// AvatarView.tsx
import { voiceServiceManager } from './services/VoiceServiceSingleton';

useEffect(() => {
  // 只检查，不初始化
  if (!voiceServiceManager.isInitialized()) {
    console.warn('Voice service not initialized yet');
  }
}, []);
```

## 🚨 问题 2: 错误处理不统一

### 当前问题

```typescript
// 有的 throw Error
throw new Error('初始化失败');

// 有的返回 false
return false;

// 有的 console.error 后继续
console.error('失败');
```

### 修复方案

**创建统一错误类：**

```typescript
// src/services/VoiceErrors.ts

export enum VoiceErrorCode {
  INIT_FAILED = 'INIT_FAILED',
  MODEL_NOT_FOUND = 'MODEL_NOT_FOUND',
  PERMISSION_DENIED = 'PERMISSION_DENIED',
  NOT_INITIALIZED = 'NOT_INITIALIZED',
  AUDIO_CONTEXT_FAILED = 'AUDIO_CONTEXT_FAILED',
  TTS_FAILED = 'TTS_FAILED',
  STT_FAILED = 'STT_FAILED',
}

export class VoiceError extends Error {
  constructor(
    public code: VoiceErrorCode,
    public service: string,
    message: string,
    public originalError?: Error
  ) {
    super(`[${service}] ${code}: ${message}`);
    this.name = 'VoiceError';
  }
  
  static initFailed(service: string, error?: Error): VoiceError {
    return new VoiceError(
      VoiceErrorCode.INIT_FAILED,
      service,
      '服务初始化失败',
      error
    );
  }
  
  static modelNotFound(modelPath: string): VoiceError {
    return new VoiceError(
      VoiceErrorCode.MODEL_NOT_FOUND,
      'ModelLoader',
      `模型文件不存在：${modelPath}`
    );
  }
  
  static permissionDenied(feature: string): VoiceError {
    return new VoiceError(
      VoiceErrorCode.PERMISSION_DENIED,
      'Permissions',
      `${feature} 权限被拒绝`
    );
  }
}

// 错误处理工具
export function handleVoiceError(
  error: unknown,
  context: string,
  fallback?: () => void
): void {
  if (error instanceof VoiceError) {
    console.error(`[Voice] ${context}:`, error);
    
    // 根据错误类型处理
    switch (error.code) {
      case VoiceErrorCode.MODEL_NOT_FOUND:
        console.warn('请运行：npm run download-models');
        break;
      case VoiceErrorCode.PERMISSION_DENIED:
        console.warn('请检查浏览器权限设置');
        break;
    }
  } else {
    console.error(`[Voice] ${context}: 未知错误`, error);
  }
  
  fallback?.();
}
```

**统一使用：**

```typescript
// UltimateVoiceService.ts
import { VoiceError, handleVoiceError } from './VoiceErrors';

async initialize(): Promise<void> {
  try {
    await localVoiceService.initialize(config);
  } catch (error) {
    handleVoiceError(
      error,
      'LocalVoice init',
      () => {
        // 降级处理
        console.warn('LocalVoice 不可用，使用基础模式');
      }
    );
    throw VoiceError.initFailed('LocalVoice', error as Error);
  }
}
```

## 🚨 问题 3: 循环依赖风险

### 当前依赖链

```
UltimateVoiceService
  ↓
LocalVoiceService
  ↓
WhisperService
  ↓
GatewayService
  ↓
useAppStore
  ↓ (可能)
UltimateVoiceService ❌
```

### 修复方案

**方案 A: 使用事件总线解耦**

```typescript
// src/services/VoiceEvents.ts
type VoiceEventHandler = (data: any) => void;

class VoiceEventBus {
  private listeners: Map<string, Set<VoiceEventHandler>> = new Map();
  
  on(event: string, handler: VoiceEventHandler): void {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, new Set());
    }
    this.listeners.get(event)!.add(handler);
  }
  
  off(event: string, handler: VoiceEventHandler): void {
    this.listeners.get(event)?.delete(handler);
  }
  
  emit(event: string, data?: any): void {
    this.listeners.get(event)?.forEach(handler => handler(data));
  }
}

export const voiceEventBus = new VoiceEventBus();

// 事件类型
export const VOICE_EVENTS = {
  INITIALIZED: 'voice:initialized',
  SPEAKING: 'voice:speaking',
  LISTENING: 'voice:listening',
  EMOTION_CHANGED: 'voice:emotion_changed',
  ERROR: 'voice:error',
} as const;
```

**使用：**
```typescript
// LocalVoiceService.ts
import { voiceEventBus, VOICE_EVENTS } from './VoiceEvents';

async sendMessage(text: string): Promise<void> {
  // 不直接依赖 GatewayService
  // 而是触发事件
  voiceEventBus.emit('voice:message', { text });
}

// GatewayService.ts
import { voiceEventBus, VOICE_EVENTS } from './VoiceEvents';

voiceEventBus.on('voice:message', (data) => {
  // 处理消息发送
  this.sendMessage(data.text);
});
```

**方案 B: 延迟依赖注入**

```typescript
// 不直接 import，而是通过参数传入
class LocalVoiceService {
  private gatewayService: any = null;
  
  setGatewayService(gateway: any): void {
    this.gatewayService = gateway;
  }
  
  async sendMessage(text: string): Promise<void> {
    if (!this.gatewayService) {
      throw new Error('GatewayService not set');
    }
    this.gatewayService.sendMessage(text);
  }
}

// 在应用入口设置依赖
localVoiceService.setGatewayService(gatewayService);
```

## 🚨 问题 4: 单例状态管理

### 问题

```typescript
// 所有组件共享同一个单例
export const ultimateVoiceService = new UltimateVoiceService();

// AvatarView 使用
// ChatPanel 也使用
// → 状态冲突
```

### 修复方案

**方案 A: 使用 React Context**

```typescript
// src/contexts/VoiceContext.tsx
import { createContext, useContext, useEffect, useState } from 'react';
import { ultimateVoiceService, UltimateState } from '../services/UltimateVoiceService';

interface VoiceContextValue {
  service: typeof ultimateVoiceService;
  state: UltimateState | null;
  isInitialized: boolean;
  error: Error | null;
}

const VoiceContext = createContext<VoiceContextValue | null>(null);

export function VoiceProvider({ children }: { children: React.ReactNode }) {
  const [state, setState] = useState<UltimateState | null>(null);
  const [isInitialized, setInitialized] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  
  useEffect(() => {
    let animationFrame: number;
    let lastTime = performance.now();
    
    const animate = (time: number) => {
      const deltaTime = (time - lastTime) / 1000;
      lastTime = time;
      
      if (isInitialized) {
        const newState = ultimateVoiceService.update(deltaTime);
        setState(newState);
      }
      
      animationFrame = requestAnimationFrame(animate);
    };
    
    animationFrame = requestAnimationFrame(animate);
    
    return () => cancelAnimationFrame(animationFrame);
  }, [isInitialized]);
  
  useEffect(() => {
    ultimateVoiceService.initialize()
      .then(() => setInitialized(true))
      .catch(err => setError(err));
  }, []);
  
  const value = {
    service: ultimateVoiceService,
    state,
    isInitialized,
    error,
  };
  
  return (
    <VoiceContext.Provider value={value}>
      {children}
    </VoiceContext.Provider>
  );
}

export function useVoice() {
  const context = useContext(VoiceContext);
  if (!context) {
    throw new Error('useVoice must be used within VoiceProvider');
  }
  return context;
}
```

**使用：**
```typescript
// main.tsx
import { VoiceProvider } from './contexts/VoiceContext';

createRoot(app!).render(
  <VoiceProvider>
    <App />
  </VoiceProvider>
);

// AvatarView.tsx
import { useVoice } from './contexts/VoiceContext';

function AvatarView() {
  const { state, isInitialized, error } = useVoice();
  
  if (error) {
    return <div>语音服务初始化失败</div>;
  }
  
  if (!isInitialized) {
    return <div>初始化中...</div>;
  }
  
  // 使用 state 渲染
  return (
    <div>
      嘴型：{state?.mouthOpen}
      瞳孔：{state?.pupilSize}
    </div>
  );
}
```
