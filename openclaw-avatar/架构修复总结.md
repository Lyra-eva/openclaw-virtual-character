## 🚀 使用方式

### 1. 应用入口 (已自动配置)

```typescript
// src/main.tsx
<VoiceProvider 
  config={{
    enableMicroExpressions: true,
    enableEmotionalTTS: true,
    enablePredictiveLipSync: true,
    enableBodyLanguage: true,
    enableContextAware: true,
  }}
  autoInitialize={true}
>
  <App />
</VoiceProvider>
```

### 2. 组件中使用

```typescript
// 任何组件中
import { useVoice } from '@/contexts/VoiceContext';

function MyComponent() {
  const { 
    state,      // 当前状态
    isReady,    // 是否就绪
    error,      // 错误信息
    speak,      // 说话方法
    startListening, // 开始监听
  } = useVoice();
  
  return (
    <div>
      {isReady ? '已就绪' : '初始化中...'}
      {error && <div>错误：{error.message}</div>}
      <button onClick={() => speak('你好')}>说话</button>
    </div>
  );
}
```

### 3. 错误处理

```typescript
import { VoiceError, handleVoiceError } from './services/VoiceErrors';

try {
  await someVoiceOperation();
} catch (error) {
  handleVoiceError(error, 'MyService', 'Operation failed');
}
```

### 4. 事件订阅

```typescript
import { voiceEventBus, VOICE_EVENTS } from './services/VoiceEvents';

voiceEventBus.on(VOICE_EVENTS.EMOTION_CHANGED, (data) => {
  console.log('情绪变化:', data);
});
```
